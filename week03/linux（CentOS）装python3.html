使用前最好用新装的虚拟机¶

python3

yum -y install zlib-devel bzip2-devel openssl-devel ncurses-devel sqlite-devel readline-devel tk-devel gdbm-devel db4-devel libpcap-devel xz-devel

cd /usr

mkdir -p /usr/software

tar -zxvf Python-3.7.12.tgz -C /usr/software/

yum install gcc

yum install libffi-devel -y

cd /usr/software/Python-3.7.12/

./configure --prefix=/usr/software/Python-3.7.12/

make

make install

ln -s /usr/software/Python-3.7.12/bin/python3 /usr/bin/python3

ln -s /usr/software/Python-3.7.12/bin/pip3 /usr/bin/pip3

python3 -Vpython

pip3 -Vpip
